


mkdir -p roles/rotate-jenkins-ssh-key/{tasks,defaults}

vim roles/rotate-jenkins-ssh-key/defaults/main.yml 
>>
---
jenkins_authorized_key_file: /home/jenkins/.ssh/authorized_keys
is_exclusive: no

vim roles/rotate-jenkins-ssh-key/tasks/main.yml 
>>
---
- name: Add SSH Key to authorized_keys file
  authorized_key:
    user: jenkins
    state: present
    manage_dir: no
    exclusive: "{{ is_exclusive }}"
    path: "{{ jenkins_authorized_key_file }}"
    key: "{{ ssh_key_contents }}"  



vim update-rotate-jenkins-key.yml
>> 
---
# Creates new local ssh key on jenkins box
- hosts: localhost
  tags: generate_new_key
  tasks:
    - name: backup public key to secrets manager
      aws_secret:
        name: jenkins/sshbackup/{{ ansible_date_time.iso8601_basic_short }}.pub
        state: present
        secret_type: 'string'
        secret: "{{ lookup('file', '~/.ssh/id_rsa.pub') }}"

    - name: backup private key to secrets manager
      aws_secret:
        name: jenkins/sshbackup/{{ ansible_date_time.iso8601_basic_short }}
        state: present
        secret_type: 'string'
        secret: "{{ lookup('file', '~/.ssh/id_rsa') }}"

    - name: Generate new ssh key
      command: ssh-keygen -t rsa -N '' -f {{ temp_location }}
      args:
        creates: "{{ temp_location }}"

# Add new key to authorized_keys file
- hosts: paz_ami_hosts
  tags: rotate_key
  become: true
  pre_tasks:
    - name: set key file fact
      set_fact:
        ssh_key_contents: "{{ lookup('file', temp_location + '.pub') }}"
      delegate_to: localhost

  roles:
    - { role: rotate-jenkins-ssh-key }

# Check that new key works
- hosts: paz_ami_hosts
  tags: rotate_key
  remote_user: jenkins
  vars:
    ansible_ssh_private_key_file: "{{ temp_location }}"
  tasks:
    - name: Check connectivity to all hosts
      command: echo hello

# Set new key with is_exclusive = true
- hosts: paz_ami_hosts
  tags: rotate_key
  become: true
  pre_tasks:
    - name: set key file fact
      set_fact:
        ssh_key_contents: "{{ lookup('file', temp_location + '.pub') }}"
      delegate_to: localhost

# remove old pub key from authorized_keys
  roles:
    - { role: rotate-jenkins-ssh-key, is_exclusive: yes }

# backup old key, and set new key on local
- hosts: localhost
  tags: generate_new_key
  tasks:
    - name: Add Public Key to AWS Secrets Manager
      community.aws.aws_secret:
          name: "ssh_public_key/jenkins"
          state: present
          secret_type: 'string'
          secret: "{{ lookup('file', temp_location + '.pub') }}"

    - name: Backup old private key
      command: mv ~/.ssh/id_rsa ~/.ssh/id_rsa.bak
      run_once: true
    - name: Backup old public key
      command: mv ~/.ssh/id_rsa.pub ~/.ssh/id_rsa.pub.bak
      run_once: true
    - name: Move new private key
      command: cp {{ temp_location }} ~/.ssh/id_rsa
      run_once: true
    - name: Move new pub key
      command: cp {{ temp_location }}.pub ~/.ssh/id_rsa.pub
      run_once: true

mkdir ~/.ssh/ORIG
cp ~/.ssh/id_rsa* ~/.ssh/ORIG
ansible-playbook update-rotate-jenkins-key.yml -e "temp_location=/tmp/ssh-migration" --tags "generate_new_key"
ansible-playbook update-rotate-jenkins-key.yml -e "temp_location=/tmp/ssh-migration" --tags "rotate_key"
ansible-playbook update-rotate-jenkins-key.yml -e "temp_location=/tmp/ssh-migration" --tags "rotate_key" --private-key ~/.ssh/ORIG/id_rsa
