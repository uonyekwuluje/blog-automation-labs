  stg_instances: "tags.Environment == 'stg1'"
  non_dev_instances: "tags.Environment != 'dev1'"
  infra_ami_hosts: "tags.ami_type == 'infra_ami'"
  paz_ami_hosts: "tags.ami_type == 'paz_ami'"
    
  ##############
  # Mongo groups
  ##############
  mongo: "inventory_hostname is search('mongo')"
  mongopipe: "inventory_hostname is search('mongopipe')"
  mongoui: "inventory_hostname is search('mongoui')"

  # Everything running mongo, server or client (except Jenkins,
  # which has a completely different setup!)
  mongo_hosts: "tags.type in ['merchant-api', 'pipeline'] or inventory_hostname is search('mongo')"
  # All-inclusive list of pipeline-related hosts with MongoDB installed. (Used for Cloud Mgr. agents.)
  mongopipe_hosts: "tags.type in ['merchant-api', 'pipeline'] or inventory_hostname is search('mongopipe')"
  # All the Mongo hosts which use LVM
  mongo_lvm_hosts: "tags.type in ['mongopipe-dbrs', 'mongoui-dbrs'] and tags.rs_type != 'arbiter' and tags.shard_zone is not defined"
  # All the Mongo hosts which store data in the /data directory
  mongo_data_hosts: "tags.type in ['mongopipe-dbrs', 'mongopipe-cfrs', 'mongoui-dbrs'] and tags.rs_type != 'arbiter'"
  mongo_data_non_config_hosts: "tags.type in ['mongopipe-dbrs', 'mongoui-dbrs'] and tags.rs_type != 'arbiter'"
  mongo_router_hosts: "tags.type in ['merchant-api', 'pipeline', 'mongopipe-router']"

  # Shard groups
  mongo_shard_hosts: "tags.type in ['mongopipe-dbrs', 'mongoui-dbrs'] and tags.shard_zone is defined"

  mongopipe_shard_hosts: "tags.type == 'mongopipe-dbrs' and tags.shard_zone is defined"
  mongoui_shard_hosts: "tags.type == 'mongoui-dbrs' and tags.shard_zone is defined"

  mongo_non_partition_shard_hosts: "tags.type in ['mongopipe-dbrs', 'mongopipe-cfrs', 'mongoui-dbrs'] and tags.shard_zone is not defined"
  mongopipe_non_partition_shard_hosts: "tags.type == 'mongopipe-dbrs' and tags.shard_zone is not defined"
  mongoui_non_partition_shard_hosts: "tags.type == 'mongoui-dbrs' and tags.shard_zone is not defined"

# Create Keys
---------------------------
aws secretsmanager create-secret --name jenkins_user_sshkey/private_key --secret-string file://id_rsa
aws secretsmanager create-secret --name jenkins_user_sshkey/public_key --secret-string file://id_rsa.pub

# Delete Keys
---------------------------
aws secretsmanager list-secrets
aws secretsmanager delete-secret --secret-id "jenkins_user_sshkey/private_key" --force-delete-without-recovery
aws secretsmanager delete-secret --secret-id "jenkins_user_sshkey/public_key" --force-delete-without-recovery
